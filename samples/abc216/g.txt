int N [1, 2*(10**5)]
int M [1, min(100, N * (N+1) // 2)]
graph<N, M> LR no_multiple_edge
col<int, M> X [1, abs(LR[1] - LR[0]) + 1]
+++
for i in range(M):
    if LR[i][0] > LR[i][1]:
        LR[i][0], LR[i][1] = LR[i][1], LR[i][0]
---
N M
LR X
